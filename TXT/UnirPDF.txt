    <script>
        // --- PDF Merge Logic ---
        const dropzone = document.getElementById("dropzone");
        const fileInput = document.getElementById("fileInput");
        const list = document.getElementById("list");
        const mergeBtn = document.getElementById("mergeBtn");
        const progressText = document.getElementById("progressText");

        // El mapa nos permite recuperar el File original a partir del elemento <li> del DOM
        const fileMap = new WeakMap();

        // Drag state para reordenar
        let dragSrcLi = null;

        dropzone.addEventListener("click", () => fileInput.click());
        dropzone.addEventListener("dragover", (e) => { e.preventDefault(); dropzone.classList.add("dragover"); });
        dropzone.addEventListener("dragleave", () => dropzone.classList.remove("dragover"));
        dropzone.addEventListener("drop", (e) => {
            e.preventDefault();
            dropzone.classList.remove("dragover");
            addFiles(e.dataTransfer.files);
        });
        fileInput.addEventListener("change", (e) => addFiles(e.target.files));

        function addFiles(files) {
            for (let file of files) {
                if (file.type !== "application/pdf") continue;

                const li = document.createElement("li");
                li.draggable = true;

                const handle = document.createElement("span");
                handle.className = "drag-handle-icon";
                handle.innerHTML = `
                    <svg viewBox="0 0 24 24" fill="currentColor">
                        <circle cx="9" cy="5" r="2" />
                        <circle cx="9" cy="12" r="2" />
                        <circle cx="9" cy="19" r="2" />
                        <circle cx="15" cy="5" r="2" />
                        <circle cx="15" cy="12" r="2" />
                        <circle cx="15" cy="19" r="2" />
                    </svg>
                `;

                const fileName = document.createElement("span");
                fileName.textContent = file.name;
                fileName.style.flex = "1";

                const deleteBtn = document.createElement("button");
                deleteBtn.innerHTML = "X";
                deleteBtn.className = "delete-btn";
                deleteBtn.title = "Eliminar";
                deleteBtn.addEventListener("click", () => list.removeChild(li));

                li.appendChild(handle);
                li.appendChild(fileName);
                li.appendChild(deleteBtn);
                list.appendChild(li);

                // Guardar referencia al File en el WeakMap
                fileMap.set(li, file);

                // Eventos drag & drop para reordenar
                li.addEventListener("dragstart", () => {
                    dragSrcLi = li;
                    setTimeout(() => li.classList.add("dragging"), 0);
                });
                li.addEventListener("dragend", () => {
                    li.classList.remove("dragging");
                    document.querySelectorAll("#list li").forEach(el => el.classList.remove("drag-over-li"));
                });
                li.addEventListener("dragover", (e) => {
                    e.preventDefault();
                    if (li !== dragSrcLi) li.classList.add("drag-over-li");
                });
                li.addEventListener("dragleave", () => li.classList.remove("drag-over-li"));
                li.addEventListener("drop", (e) => {
                    e.stopPropagation();
                    li.classList.remove("drag-over-li");
                    if (dragSrcLi && dragSrcLi !== li) {
                        const rect = li.getBoundingClientRect();
                        const midY = rect.top + rect.height / 2;
                        if (e.clientY < midY) {
                            list.insertBefore(dragSrcLi, li);
                        } else {
                            list.insertBefore(dragSrcLi, li.nextSibling);
                        }
                    }
                });

                // --- Soporte para dispositivos táctiles ---
                li.addEventListener("touchstart", onTouchStart, { passive: false });
                li.addEventListener("touchmove", onTouchMove, { passive: false });
                li.addEventListener("touchend", onTouchEnd);
            }
        }

        // Unir PDFs respetando el orden actual del DOM
        mergeBtn.addEventListener("click", async () => {
            const listItems = Array.from(list.querySelectorAll("li"));
            if (listItems.length === 0) {
                alert("No has añadido ningún PDF.");
                return;
            }

            mergeBtn.disabled = true;
            progressText.style.display = "block";

            try {
                const mergedPdf = await PDFLib.PDFDocument.create();
                const total = listItems.length;

                for (let i = 0; i < total; i++) {
                    const file = fileMap.get(listItems[i]);
                    progressText.textContent = `Procesando archivo ${i + 1} de ${total}: ${file.name}`;

                    const arrayBuffer = await file.arrayBuffer();
                    const pdf = await PDFLib.PDFDocument.load(arrayBuffer);
                    const pages = await mergedPdf.copyPages(pdf, pdf.getPageIndices());
                    pages.forEach(p => mergedPdf.addPage(p));
                }

                progressText.textContent = "Generando PDF final...";
                const mergedBytes = await mergedPdf.save();

                const blob = new Blob([mergedBytes], { type: "application/pdf" });
                const url = URL.createObjectURL(blob);
                const a = document.createElement("a");
                a.href = url;
                a.download = "PDF_Unido.pdf";
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
                URL.revokeObjectURL(url);

                progressText.textContent = "¡PDF unido descargado con éxito!";
                setTimeout(() => { progressText.style.display = "none"; }, 3000);

            } catch (err) {
                alert("Error al unir los PDFs.");
                console.error(err);
                progressText.textContent = "Error en el proceso.";
            } finally {
                mergeBtn.disabled = false;
            }
        });

        // --- Soporte para dispositivos táctiles ---
        function onTouchStart(e) {
            dragSrcLi = this;
            // No prevenimos default aquí para permitir scroll si no se toca el handle
            // (aunque el handle tiene touch-action: none en CSS)
            setTimeout(() => this.classList.add("dragging"), 0);
        }

        function onTouchMove(e) {
            if (!dragSrcLi) return;

            // Prevenir scroll si estamos arrastrando
            e.preventDefault();

            const touch = e.touches[0];
            const target = document.elementFromPoint(touch.clientX, touch.clientY);

            // Subir por el DOM hasta encontrar un LI
            let liTarget = target;
            while (liTarget && liTarget.tagName !== "LI") {
                liTarget = liTarget.parentElement;
            }

            if (liTarget && liTarget.tagName === "LI" && liTarget !== dragSrcLi) {
                const rect = liTarget.getBoundingClientRect();
                const midY = rect.top + rect.height / 2;

                if (touch.clientY < midY) {
                    list.insertBefore(dragSrcLi, liTarget);
                } else {
                    list.insertBefore(dragSrcLi, liTarget.nextSibling);
                }
            }
        }

        function onTouchEnd() {
            if (dragSrcLi) {
                dragSrcLi.classList.remove("dragging");
                dragSrcLi = null;
            }
        }

    </script>
